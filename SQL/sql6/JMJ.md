# 1. 조건에 맞는 도서와 저자 리스트 출력
```mysql
SELECT B.BOOK_ID, A.AUTHOR_NAME, DATE_FORMAT(B.PUBLISHED_DATE, "%Y-%m-%d") AS PUBLISHED_DATE
FROM AUTHOR AS A JOIN BOOK AS B
ON A.AUTHOR_ID = B.AUTHOR_ID
WHERE B.CATEGORY = '경제'
ORDER BY B.PUBLISHED_DATE
```

# 2. 5월 식품들의 총매출 조회
```mysql
SELECT A.PRODUCT_ID, A.PRODUCT_NAME, A.PRICE * B.COUNT AS TOTAL_SALES
FROM FOOD_PRODUCT AS A 
JOIN
(SELECT PRODUCT_ID, SUM(AMOUNT) AS COUNT 
 FROM FOOD_ORDER 
 WHERE DATE_FORMAT(PRODUCE_DATE, "%Y-%m-%d") BETWEEN '2022-05-01' AND '2022-05-31' 
 GROUP BY PRODUCT_ID) AS B
ON A.PRODUCT_ID = B.PRODUCT_ID
ORDER BY TOTAL_SALES DESC, A.PRODUCT_ID
```

# 3. 그룹별 조건에 맞는 식당 목록 출력
```mysql
SELECT A.MEMBER_NAME, B.REVIEW_TEXT, DATE_FORMAT(B.REVIEW_DATE, "%Y-%m-%d") AS REVIEW_DATE
FROM MEMBER_PROFILE AS A JOIN REST_REVIEW AS B ON A.MEMBER_ID = B.MEMBER_ID
WHERE A.MEMBER_ID = 
(SELECT MEMBER_ID FROM REST_REVIEW GROUP BY MEMBER_ID ORDER BY COUNT(MEMBER_ID) DESC LIMIT 1)
ORDER BY REVIEW_DATE, B.REVIEW_TEXT

```

# 4. 주문량이 많은 아이스크림들 조회
```mysql
SELECT A.FLAVOR
FROM FIRST_HALF AS A JOIN JULY AS B
ON A.FLAVOR = B.FLAVOR
GROUP BY A.FLAVOR
ORDER BY A.TOTAL_ORDER + SUM (B.TOTAL_ORDER) DESC
LIMIT 3
```

# 5. 보호소에서 중성화한 동물
```mysql
SELECT INS.ANIMAL_ID, INS.ANIMAL_TYPE, INS.NAME
FROM ANIMAL_INS AS INS JOIN ANIMAL_OUTS AS OUTS
                            ON INS.ANIMAL_ID = OUTS.ANIMAL_ID
WHERE INS.SEX_UPON_INTAKE != OUTS.SEX_UPON_OUTCOME
```

# 6. 상품을 구매한 회원 비율
```mysql
SELECT YEAR, MONTH, COUNT(*) AS PUCHASED_USERS, ROUND((COUNT(*)/ (SELECT COUNT(*) FROM USER_INFO WHERE YEAR(JOINED) = 2021)), 1) AS PUCHASED_RATIO
FROM (
    SELECT DISTINCT YEAR(S.SALES_DATE) AS YEAR, MONTH(S.SALES_DATE) AS MONTH, U.USER_ID
    FROM ONLINE_SALE S
    JOIN USER_INFO U ON S.USER_ID = U.USER_ID AND YEAR(JOINED) = 2021
) A
GROUP BY YEAR, MONTH
ORDER BY YEAR, MONTH
```
